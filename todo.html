it("Works for async failures.", function (done) {
  // Wait for results.
  setTimeout(function () {
    true.should.equal(false);
    return;

    var err;
    try {
      true.should.equal(false);
    } catch (error) {
      console.log(typeof error, error, error.stack);
      err = new Error(error);
      err.stack = error.stack;

    } finally {
      done(err);
    }
  }, 100);
});
